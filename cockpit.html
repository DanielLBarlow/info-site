Cockpit install

  1. sudo apt update
  2. sudo apt install cockpit -y
  3. sudo systemctl start cockpit
  4. sudo systemctl status

firewalld install

  1. sudo apt update
  2. sudo apt install firewalld
  3. sudo systemctl enable firewalld
  4. sudo systemctl start firewalld
  5. sudo firewall-cmd --state
  6. sudo firewall-cmd --add-service-cockpit
  7. sudo firewall-cmd --add-service-cockpit --permanent
  8. pkcheck --action-id org.fedoraproject.FirewallD1.all --process $$ --allow-user-interaction
  9. pkcheck --action-id org.fedoraproject.FirewallD1.all --process $$ --allow-user-interaction --enable-internal-agent
  10. sudo reboot now

Software Updates Fix

  1. sudo nano /etc/netplan/00-installer-config.yaml
  2.                 
                network:
                  renderer: NetworkManager
                  ethernets:
                    enp3s0:
                      dhcp4: true
                 version: 2
  3. Press CTRL + X then hit Y. Then hit Enter.
  4. sudo netplan try
  5. sudo netplan apply
